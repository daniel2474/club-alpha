SELECT * FROM CA_APARTADOS WHERE 
Id_apartados in(select id_apartados from clases where CLUB='Club Alpha 2') AND
DIA::DATE BETWEEN '2022-06-09'::DATE AND '2022-06-12'::DATE AND ACTIVO IS FALSE
and updated between '2022-06-06 00:00:00' and '2022-06-06 23:59:59'



update ca_apartados set activo=false
where id_apartados in(SELECT id_apartados FROM CLASES WHERE CLUB='Club Alpha 2' AND SEGMENTACION IS TRUE AND LUGAR='CANCHA DE TENIS 4'
AND DIA BETWEEN '2022-06-20' AND '2022-06-30' ORDER BY DIA,RANGO)
